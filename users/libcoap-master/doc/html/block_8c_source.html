00082     debug("illegal block requested\n");

00086   avail = pdu->max_size - pdu->length - 4;
00087   want = 1 << (block->szx + 4);
00088 
00089   /* check if entire block fits in message */
00090   if (want <= avail) {
00091     block->m = want < data_length - start;
00092   } else {
00093     /* Sender has requested a block that is larger than the remaining
00094      * space in pdu. This is ok if the remaining data fits into the pdu
00095      * anyway. The block size needs to be adjusted only if there is more
00096      * data left that cannot be delivered in this message. */
00097 
00098     if (data_length - start <= avail) {
00099 
00100       /* it's the final block and everything fits in the message */
00101       block->m = 0;
00102     } else {
00103       unsigned char szx;
00104 
00105       /* we need to decrease the block size */
00106       if (avail < 16) {     /* bad luck, this is the smallest block size */
00107     debug("not enough space, even the smallest block does not fit");
00108     return -3;
00109       }
00110       debug("decrease block size for %zu to %d\n", avail, coap_fls(avail) - 5);
00111       szx = block->szx;
00112       block->szx = coap_fls(avail) - 5;
00113       block->m = 1;
