00179 unsigned int
00180 coap_adjust_basetime(coap_context_t *ctx, coap_tick_t now) {
00181   unsigned int result = 0;
00182   coap_tick_diff_t delta = now - ctx->sendqueue_basetime;
00183 
00184   if (ctx->sendqueue) {
00185     /* delta < 0 means that the new time stamp is before the old. */
00186     if (delta <= 0) {
00187       ctx->sendqueue->t -= delta;
00188     } else {
00189       /* This case is more complex: The time must be advanced forward,
00190        * thus possibly leading to timed out elements at the queue's
00191        * start. For every element that has timed out, its relative
00192        * time is set to zero and the result counter is increased. */
00193 
00194       coap_queue_t *q = ctx->sendqueue;
00195       coap_tick_t t = 0;
00196       while (q && (t + q->t < (coap_tick_t)delta)) {
00197     t += q->t;
00198     q->t = 0;
00199     result++;
00200     q = q->next;
00201       }
00202 
00203       /* finally adjust the first element that has not expired */
00204       if (q) {
00205     q->t = (coap_tick_t)delta - t;
00206       }
00207     }
00208   }
00209 
00210   /* adjust basetime */
00211   ctx->sendqueue_basetime += delta;
00212 
00213   return result;
00214 }

00217 coap_insert_node(coap_queue_t **queue, coap_queue_t *node) {
00218   coap_queue_t *p, *q;
00219   if ( !queue || !node )
00220     return 0;
00221 
00222   /* set queue head if empty */
00223   if ( !*queue ) {
00224     *queue = node;
00225     return 1;
00226   }
00227 
00228   /* replace queue head if PDU's time is less than head's time */
00229   q = *queue;
00230   if (node->t < q->t) {
00231     node->next = q;
00232     *queue = node;
00233     q->t -= node->t;        /* make q->t relative to node->t */
00234     return 1;
00235   }
00236 
00237   /* search for right place to insert */
00238   do {
00239     node->t -= q->t;        /* make node-> relative to q->t */
00240     p = q;
00241     q = q->next;
00242   } while (q && q->t <= node->t);
00243 
00244   /* insert new item */
00245   if (q) {
00246     q->t -= node->t;        /* make q->t relative to node->t */
00247   }
00248   node->next = q;
00249   p->next = node;
00250   return 1;

00672 static inline unsigned int
00673 calc_timeout(unsigned char r) {
00674   unsigned int result;
00675 
00676   /* The integer 1.0 as a Qx.FRAC_BITS */
00677 #define FP1 Q(FRAC_BITS, 1)
00678 
00679   /* rounds val up and right shifts by frac positions */
00680 #define SHR_FP(val,frac) (((val) + (1 << ((frac) - 1))) >> (frac))
00681 
00682   /* Inner term: multiply ACK_RANDOM_FACTOR by Q0.MAX_BITS[r] and
00683    * make the result a rounded Qx.FRAC_BITS */
00684   result = SHR_FP((ACK_RANDOM_FACTOR - FP1) * r, MAX_BITS);
00685 
00686   /* Add 1 to the inner term and multiply with ACK_TIMEOUT, then
00687    * make the result a rounded Qx.FRAC_BITS */
00688   result = SHR_FP(((result + FP1) * ACK_TIMEOUT), FRAC_BITS);
00689 
00690   /* Multiply with COAP_TICKS_PER_SECOND to yield system ticks
00691    * (yields a Qx.FRAC_BITS) and shift to get an integer */
00692   return SHR_FP((COAP_TICKS_PER_SECOND * result), FRAC_BITS);

00702             const coap_endpoint_t *local_interface,
00703             const coap_address_t *dst,
00704             coap_pdu_t *pdu) {

00706   coap_tick_t now;
00707   unsigned char r;
00708

00713   }

00716   if (COAP_INVALID_TID == node->id) {
00717     debug("coap_send_confirmed: error sending pdu\n");
00718     coap_free_node(node);
00719     return COAP_INVALID_TID;
00720   }
00721   
00722   prng((unsigned char *)&r,sizeof(r));
00723 
00724   /* add timeout in range [ACK_TIMEOUT...ACK_TIMEOUT * ACK_RANDOM_FACTOR] */
00725   node->timeout = calc_timeout(r);
00726 
00727   node->local_if = *local_interface;

00737    * the retransmission queue, the base time is set to the current
00738    * time and the retransmission time is node->timeout. If there is
00739    * already an entry in the sendqueue, we must check if this node is
00740    * to be retransmitted earlier. Therefore, node->timeout is first
00741    * normalized to the base time and then inserted into the queue with
00742    * an adjusted relative time.
00743    */
00744   coap_ticks(&now);

00751   }

00773 #endif /* WITH_CONTIKI */

00779 coap_retransmit(coap_context_t *context, coap_queue_t *node) {

00784   if (node->retransmit_cnt < COAP_DEFAULT_MAX_RETRANSMIT) {

00794       node->retransmit_cnt, ntohs(node->pdu->hdr->id));

00797                   &node->remote, node->pdu);

00803 #ifndef WITH_CONTIKI

00939   coap_queue_t *p, *q;
00940 
00941   if ( !queue || !*queue)
00942     return 0;
00943 
00944   /* replace queue head if PDU's time is less than head's time */
00945 
00946   if ( id == (*queue)->id ) { /* found transaction */
00947     *node = *queue;
00948     *queue = (*queue)->next;
00949     if (*queue) {     /* adjust relative time of new queue head */

00954     debug("*** removed transaction %u\n", id);

00959   q = *queue;

00962     q = q->next;

00966     p->next = q->next;
00967     if (p->next) {      /* must update relative time of p->next */
00968       p->next->t += q->t;

01035 coap_pdu_t *
01036 coap_new_error_response(coap_pdu_t *request, unsigned char code, 
01037             coap_opt_filter_t opts) {
01038   coap_opt_iterator_t opt_iter;
01039   coap_pdu_t *response;
01040   size_t size = sizeof(coap_hdr_t) + request->hdr->token_length;
01041   int type; 
01042   coap_opt_t *option;
01043   unsigned short opt_type = 0;  /* used for calculating delta-storage */
01044 
01045 #if COAP_ERROR_PHRASE_LENGTH > 0
01046   char *phrase = coap_response_phrase(code);
01047 
01048   /* Need some more space for the error phrase and payload start marker */
01049   if (phrase)
01050     size += strlen(phrase) + 1;
01051 #endif
01052 
01053   assert(request);
01054 
01055   /* cannot send ACK if original request was not confirmable */
01056   type = request->hdr->type == COAP_MESSAGE_CON 
01057     ? COAP_MESSAGE_ACK
01058     : COAP_MESSAGE_NON;
01059 
01060   /* Estimate how much space we need for options to copy from
01061    * request. We always need the Token, for 4.02 the unknown critical
01062    * options must be included as well. */
01063   coap_option_clrb(opts, COAP_OPTION_CONTENT_TYPE); /* we do not want this */
01064 
01065   coap_option_iterator_init(request, &opt_iter, opts);
01066 
01067   /* Add size of each unknown critical option. As known critical
01068      options as well as elective options are not copied, the delta
01069      value might grow.
01070    */
01071   while((option = coap_option_next(&opt_iter))) {
01072     unsigned short delta = opt_iter.type - opt_type;
01073     /* calculate space required to encode (opt_iter.type - opt_type) */
01074     if (delta < 13) {
01075       size++;
01076     } else if (delta < 269) {
01077       size += 2;
01078     } else {
01079       size += 3;
01080     }
01081 
01082     /* add coap_opt_length(option) and the number of additional bytes
01083      * required to encode the option length */
01084     
01085     size += coap_opt_length(option);
01086     switch (*option & 0x0f) {
01087     case 0x0e:
01088       size++;
01089       /* fall through */
01090     case 0x0d:
01091       size++;
01092       break;
01093     default:
01094       ;
01095     }
01096 
01097     opt_type = opt_iter.type;
01098   }
01099 
01100   /* Now create the response and fill with options and payload data. */
01101   response = coap_pdu_init(type, code, request->hdr->id, size);
01102   if (response) {
01103     /* copy token */
01104     if (!coap_add_token(response, request->hdr->token_length, 
01105             request->hdr->token)) {
01106       debug("cannot add token to error response\n");
01107       coap_delete_pdu(response);
01108       return NULL;
01109     }
01110 
01111     /* copy all options */
01112     coap_option_iterator_init(request, &opt_iter, opts);
01113     while((option = coap_option_next(&opt_iter)))
01114       coap_add_option(response, opt_iter.type, 
01115               COAP_OPT_LENGTH(option),
01116               COAP_OPT_VALUE(option));
01117 
01118 #if COAP_ERROR_PHRASE_LENGTH > 0
01119     /* note that diagnostic messages do not need a Content-Format option. */
01120     if (phrase)
01121       coap_add_data(response, strlen(phrase), (unsigned char *)phrase);
01122 #endif
01123   }
01124 
01125   return response;
01126 }

01150 coap_pdu_t *
01151 coap_wellknown_response(coap_context_t *context, coap_pdu_t *request) {
01152   coap_pdu_t *resp;
01153   coap_opt_iterator_t opt_iter;
01154   size_t len, wkc_len;
01155   unsigned char buf[2];
01156   int result = 0;
01157   int need_block2 = 0;     /* set to 1 if Block2 option is required */
01158   coap_block_t block;
01159   coap_opt_t *query_filter;
01160   size_t offset = 0;
01161 
01162   resp = coap_pdu_init(request->hdr->type == COAP_MESSAGE_CON 
01163                ? COAP_MESSAGE_ACK 
01164                : COAP_MESSAGE_NON,
01165                COAP_RESPONSE_CODE(205),
01166                request->hdr->id, COAP_MAX_PDU_SIZE);
01167   if (!resp) {
01168     debug("coap_wellknown_response: cannot create PDU\n");
01169     return NULL;
01170   }
01171   
01172   if (!coap_add_token(resp, request->hdr->token_length, request->hdr->token)) {
01173     debug("coap_wellknown_response: cannot add token\n");
01174     goto error;
01175   }
01176 
01177   query_filter = coap_check_option(request, COAP_OPTION_URI_QUERY, &opt_iter);
01178   wkc_len = get_wkc_len(context, query_filter);
01179 
01180   /* The value of some resources is undefined and get_wkc_len will return 0.*/
01181   if (wkc_len == 0){
01182     debug("coap_wellknown_response: undefined resource\n");
01183     /* set error code 4.00 Bad Request*/
01184     resp->hdr->code = COAP_RESPONSE_CODE(400);
01185     resp->length = sizeof(coap_hdr_t) + resp->hdr->token_length;
01186     return resp;
01187   }
01188 
01189   /* check whether the request contains the Block2 option */
01190   if (coap_get_block(request, COAP_OPTION_BLOCK2, &block)) {
01191     debug("create block\n");
01192     offset = block.num << (block.szx + 4);
01193     if (block.szx > 6) {  /* invalid, MUST lead to 4.00 Bad Request */
01194       resp->hdr->code = COAP_RESPONSE_CODE(400);
01195       return resp;
01196     } else if (block.szx > COAP_MAX_BLOCK_SZX) {
01197       block.szx = COAP_MAX_BLOCK_SZX;
01198       block.num = offset >> (block.szx + 4);
01199     }
01200 
01201     need_block2 = 1;
01202   }
01203 
01204   /* Check if there is sufficient space to add Content-Format option 
01205    * and data. We do this before adding the Content-Format option to
01206    * avoid sending error responses with that option but no actual
01207    * content. */
01208   if (resp->max_size <= (size_t)resp->length + 3) {
01209     debug("coap_wellknown_response: insufficient storage space\n");
01210     goto error;
01211   }
01212 
01213   /* Add Content-Format. As we have checked for available storage,
01214    * nothing should go wrong here. */
01215   assert(coap_encode_var_bytes(buf, 
01216             COAP_MEDIATYPE_APPLICATION_LINK_FORMAT) == 1);
01217   coap_add_option(resp, COAP_OPTION_CONTENT_FORMAT,
01218           coap_encode_var_bytes(buf, 
01219             COAP_MEDIATYPE_APPLICATION_LINK_FORMAT), buf);
01220 
01221   /* check if Block2 option is required even if not requested */
01222   if (!need_block2 && (resp->max_size - (size_t)resp->length < wkc_len)) {
01223     assert(resp->length <= resp->max_size);
01224     const size_t payloadlen = resp->max_size - resp->length;
01225     /* yes, need block-wise transfer */
01226     block.num = 0;
01227     block.m = 0;      /* the M bit is set by coap_write_block_opt() */
01228     block.szx = COAP_MAX_BLOCK_SZX;
01229     while (payloadlen < SZX_TO_BYTES(block.szx)) {
01230       if (block.szx == 0) {
01231     debug("coap_wellknown_response: message to small even for szx == 0\n");
01232     goto error;
01233       } else {
01234     block.szx--;
01235       }
01236     }
01237 
01238     need_block2 = 1;
01239   }
01240 
01241   /* write Block2 option if necessary */
01242   if (need_block2) {
01243     if (coap_write_block_opt(&block, COAP_OPTION_BLOCK2, resp, wkc_len) < 0) {
01244       debug("coap_wellknown_response: cannot add Block2 option\n");
01245       goto error;
01246     }
01247   }
01248 
01249   /* Manually set payload of response to let print_wellknown() write,
01250    * into our buffer without copying data. */
01251 
01252   resp->data = (unsigned char *)resp->hdr + resp->length;
01253   *resp->data = COAP_PAYLOAD_START;
01254   resp->data++;
01255   resp->length++;
01256   len = need_block2 ? SZX_TO_BYTES(block.szx) : resp->max_size - resp->length;
01257 
01258   result = coap_print_wellknown(context, resp->data, &len, offset, query_filter);
01259   if ((result & COAP_PRINT_STATUS_ERROR) != 0) {
01260     debug("coap_print_wellknown failed\n");
01261     goto error;
01262   } 
01263   
01264   resp->length += COAP_PRINT_OUTPUT_LENGTH(result);
01265   return resp;
01266 
01267  error:
01268   /* set error code 5.03 and remove all options and data from response */
01269   resp->hdr->code = COAP_RESPONSE_CODE(503);
01270   resp->length = sizeof(coap_hdr_t) + resp->hdr->token_length;
01271   return resp;
01272 }

01450       coap_opt_iterator_t opt_iter;

01459         coap_decode_var_bytes(coap_opt_value(observe),

01480

01482       if (respond != RESPONSE_DROP) {
01483         if (observe && (COAP_RESPONSE_CLASS(response->hdr->code) > 2)) {
01484           coap_log(LOG_DEBUG, "removed observer\n");
01485           coap_delete_observer(resource,  &node->remote, &token);

01490        * this is an empty ACK with a token, which is a malformed

01495     response->hdr->token_length = 0;
01496     response->length = sizeof(coap_hdr_t);
01497       }

01506               &node->remote, response) == COAP_INVALID_TID) {
01507       debug("cannot send response for message %d\n", node->pdu->hdr->id);
01508       }
01509       }
01510       }
01511       coap_delete_pdu(response);
01512       response = NULL;
01513     } else {
01514       warn("cannot generate response\r\n");
01515     }
01516   } else {
01517     if (WANT_WKC(node->pdu, key)) {
01518       debug("create default response for %s\n", COAP_DEFAULT_URI_WELLKNOWN);
01519       response = coap_wellknown_response(context, node->pdu);
01520       debug("have wellknown response %p\n", (void *)response);
01521     } else
01522       response = coap_new_error_response(node->pdu, COAP_RESPONSE_CODE(405), 
01523                      opt_filter);
01524 
01525     if (response && (no_response(node->pdu, response) != RESPONSE_DROP)) {
01526       if (coap_send(context, &node->local_if, &node->remote,
01527                     response) == COAP_INVALID_TID) {
01528         debug("cannot send response for transaction %u\n", node->id);
01529       }
01530     }
01531     coap_delete_pdu(response);
01532     response = NULL;
01533   }
01534
