00212 /* delete "delptr" from the hash table.
00213  * "the usual" patch-up process for the app-order doubly-linked-list.
00214  * The use of _hd_hh_del below deserves special explanation.
00215  * These used to be expressed using (delptr) but that led to a bug
00216  * if someone used the same symbol for the head and deletee, like
00217  *  HASH_DELETE(hh,users,users);
00218  * We want that to work, but by changing the head (users) below
00219  * we were forfeiting our ability to further refer to the deletee (users)
00220  * in the patch-up process. Solution: use scratch space to
00221  * copy the deletee pointer, then the latter references are via that
00222  * scratch pointer rather than through the repointed (users) symbol.
00223  */
00224 #define HASH_DELETE(hh,head,delptr)                                              \
00225 do {                                                                             \
00226     struct UT_hash_handle *_hd_hh_del;                                           \
00227     if ( ((delptr)->hh.prev == NULL) && ((delptr)->hh.next == NULL) )  {         \
00228         uthash_free((head)->hh.tbl->buckets,                                     \
00229                     (head)->hh.tbl->num_buckets*sizeof(struct UT_hash_bucket) ); \
00230         HASH_BLOOM_FREE((head)->hh.tbl);                                         \
00231         uthash_free((head)->hh.tbl, sizeof(UT_hash_table));                      \
00232         head = NULL;                                                             \
00233     } else {                                                                     \
00234         unsigned _hd_bkt;                                                        \
00235         _hd_hh_del = &((delptr)->hh);                                            \
00236         if ((delptr) == ELMT_FROM_HH((head)->hh.tbl,(head)->hh.tbl->tail)) {     \
00237             (head)->hh.tbl->tail =                                               \
00238                 (UT_hash_handle*)((ptrdiff_t)((delptr)->hh.prev) +               \
00239                 (head)->hh.tbl->hho);                                            \
00240         }                                                                        \
00241         if ((delptr)->hh.prev) {                                                 \
00242             ((UT_hash_handle*)((ptrdiff_t)((delptr)->hh.prev) +                  \
00243                     (head)->hh.tbl->hho))->next = (delptr)->hh.next;             \
00244         } else {                                                                 \
00245             DECLTYPE_ASSIGN(head,(delptr)->hh.next);                             \
00246         }                                                                        \
00247         if (_hd_hh_del->next) {                                                  \
00248             ((UT_hash_handle*)((ptrdiff_t)_hd_hh_del->next +                     \
00249                     (head)->hh.tbl->hho))->prev =                                \
00250                     _hd_hh_del->prev;                                            \
00251         }                                                                        \
00252         HASH_TO_BKT( _hd_hh_del->hashv, (head)->hh.tbl->num_buckets, _hd_bkt);   \
00253         HASH_DEL_IN_BKT(hh,(head)->hh.tbl->buckets[_hd_bkt], _hd_hh_del);        \
00254         (head)->hh.tbl->num_items--;                                             \
00255     }                                                                            \
00256     HASH_FSCK(hh,head);                                                          \
00257 } while (0)

00283  * This is for uthash developer only; it compiles away if HASH_DEBUG isn't defined.

00287 #define HASH_FSCK(hh,head)                                                       \
00288 do {                                                                             \
00289     struct UT_hash_handle *_thh;                                                 \
00290     if (head) {                                                                  \
00291         unsigned _bkt_i;                                                         \
00292         unsigned _count;                                                         \
00293         char *_prev;                                                             \
00294         _count = 0;                                                              \
00295         for( _bkt_i = 0; _bkt_i < (head)->hh.tbl->num_buckets; _bkt_i++) {       \
00296             unsigned _bkt_count = 0;                                             \
00297             _thh = (head)->hh.tbl->buckets[_bkt_i].hh_head;                      \
00298             _prev = NULL;                                                        \
00299             while (_thh) {                                                       \
00300                if (_prev != (char*)(_thh->hh_prev)) {                            \
00301                    HASH_OOPS("invalid hh_prev %p, actual %p\n",                  \
00302                     _thh->hh_prev, _prev );                                      \
00303                }                                                                 \
00304                _bkt_count++;                                                     \
00305                _prev = (char*)(_thh);                                            \
00306                _thh = _thh->hh_next;                                             \
00307             }                                                                    \
00308             _count += _bkt_count;                                                \
00309             if ((head)->hh.tbl->buckets[_bkt_i].count !=  _bkt_count) {          \
00310                HASH_OOPS("invalid bucket count %u, actual %u\n",                 \
00311                 (head)->hh.tbl->buckets[_bkt_i].count, _bkt_count);              \
00312             }                                                                    \
00313         }                                                                        \

00317         }                                                                        \

00651 /* Bucket expansion has the effect of doubling the number of buckets
00652  * and redistributing the items into the new buckets. Ideally the
00653  * items will distribute more or less evenly into the new buckets
00654  * (the extent to which this is true is a measure of the quality of
00655  * the hash function as it applies to the key domain).
00656  *
00657  * With the items distributed into more buckets, the chain length
00658  * (item count) in each bucket is reduced. Thus by expanding buckets
00659  * the hash keeps a bound on the chain length. This bounded chain
00660  * length is the essence of how a hash provides constant time lookup.
00661  *
00662  * The calculation of tbl->ideal_chain_maxlen below deserves some
00663  * explanation. First, keep in mind that we're calculating the ideal
00664  * maximum chain length based on the *new* (doubled) bucket count.
00665  * In fractions this is just n/b (n=number of items,b=new num buckets).
00666  * Since the ideal chain length is an integer, we want to calculate
00667  * ceil(n/b). We don't depend on floating point arithmetic in this
00668  * hash, so to calculate ceil(n/b) with integers we could write
00669  *
00670  *      ceil(n/b) = (n/b) + ((n%b)?1:0)
00671  *
00672  * and in fact a previous version of this hash did just that.
00673  * But now we have improved things a bit by recognizing that b is
00674  * always a power of two. We keep its base 2 log handy (call it lb),
00675  * so now we can write this with a bit shift and logical AND:
00676  *
00677  *      ceil(n/b) = (n>>lb) + ( (n & (b-1)) ? 1:0)
00678  *
00679  */
00680 #define HASH_EXPAND_BUCKETS(tbl)                                                 \
00681 do {                                                                             \
00682     unsigned _he_bkt;                                                            \
00683     unsigned _he_bkt_i;                                                          \
00684     struct UT_hash_handle *_he_thh, *_he_hh_nxt;                                 \
00685     UT_hash_bucket *_he_new_buckets, *_he_newbkt;                                \
00686     _he_new_buckets = (UT_hash_bucket*)uthash_malloc(                            \
00687              2 * tbl->num_buckets * sizeof(struct UT_hash_bucket));              \
00688     if (!_he_new_buckets) { uthash_fatal( "out of memory"); }                    \
00689     memset(_he_new_buckets, 0,                                                   \
00690             2 * tbl->num_buckets * sizeof(struct UT_hash_bucket));               \
00691     tbl->ideal_chain_maxlen =                                                    \
00692        (tbl->num_items >> (tbl->log2_num_buckets+1)) +                           \
00693        ((tbl->num_items & ((tbl->num_buckets*2)-1)) ? 1 : 0);                    \
00694     tbl->nonideal_items = 0;                                                     \
00695     for(_he_bkt_i = 0; _he_bkt_i < tbl->num_buckets; _he_bkt_i++)                \
00696     {                                                                            \
00697         _he_thh = tbl->buckets[ _he_bkt_i ].hh_head;                             \
00698         while (_he_thh) {                                                        \
00699            _he_hh_nxt = _he_thh->hh_next;                                        \
00700            HASH_TO_BKT( _he_thh->hashv, tbl->num_buckets*2, _he_bkt);            \
00701            _he_newbkt = &(_he_new_buckets[ _he_bkt ]);                           \
00702            if (++(_he_newbkt->count) > tbl->ideal_chain_maxlen) {                \
00703              tbl->nonideal_items++;                                              \
00704              _he_newbkt->expand_mult = _he_newbkt->count /                       \
00705                                         tbl->ideal_chain_maxlen;                 \
00706            }                                                                     \
00707            _he_thh->hh_prev = NULL;                                              \
00708            _he_thh->hh_next = _he_newbkt->hh_head;                               \
00709            if (_he_newbkt->hh_head) _he_newbkt->hh_head->hh_prev =               \
00710                 _he_thh;                                                         \
00711            _he_newbkt->hh_head = _he_thh;                                        \
00712            _he_thh = _he_hh_nxt;                                                 \
00713         }                                                                        \
00714     }                                                                            \
00715     uthash_free( tbl->buckets, tbl->num_buckets*sizeof(struct UT_hash_bucket) ); \
00716     tbl->num_buckets *= 2;                                                       \
00717     tbl->log2_num_buckets++;                                                     \
00718     tbl->buckets = _he_new_buckets;                                              \
00719     tbl->ineff_expands = (tbl->nonideal_items > (tbl->num_items >> 1)) ?         \
00720         (tbl->ineff_expands+1) : 0;                                              \

00897    unsigned count;

00899    /* expand_mult is normally set to 0. In this situation, the max chain length
00900     * threshold is enforced at its default value, HASH_BKT_CAPACITY_THRESH. (If
00901     * the bucket's chain exceeds this length, bucket expansion is triggered).
00902     * However, setting expand_mult to a non-zero value delays bucket expansion
00903     * (that would be triggered by additions to this particular bucket)
00904     * until its chain length reaches a *multiple* of HASH_BKT_CAPACITY_THRESH.
00905     * (The multiplier is simply expand_mult+1). The whole idea of this
00906     * multiplier is to reduce bucket expansions, since they are expensive, in
00907     * situations where we know that a particular bucket tends to be overused.
00908     * It is better to let its chain length grow to a longer yet-still-bounded
00909     * value, than to do an O(n) bucket expansion too often.
00910     */
