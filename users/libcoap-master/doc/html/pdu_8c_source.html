00343 int
00344 coap_pdu_parse(unsigned char *data, size_t length, coap_pdu_t *pdu) {
00345   coap_opt_t *opt;
00346 
00347   assert(data);
00348   assert(pdu);
00349 
00350   if (pdu->max_size < length) {
00351     debug("insufficient space to store parsed PDU\n");
00352     return 0;
00353   }
00354 
00355   if (length < sizeof(coap_hdr_t)) {
00356     debug("discarded invalid PDU\n");
00357   }
00358 
00359 #ifdef WITH_LWIP
00360   /* this verifies that with the classical copy-at-parse-time and lwip's
00361    * zerocopy-into-place approaches, both share the same idea of destination
00362    * addresses */
00363   LWIP_ASSERT("coap_pdu_parse with unexpected addresses", data == (void*)pdu->hdr);
00364   LWIP_ASSERT("coap_pdu_parse with unexpected length", length == pdu->length);
00365 #else
00366 
00367   pdu->hdr->version = data[0] >> 6;
00368   pdu->hdr->type = (data[0] >> 4) & 0x03;
00369   pdu->hdr->token_length = data[0] & 0x0f;
00370   pdu->hdr->code = data[1];
00371 #endif
00372   pdu->data = NULL;
00373 
00374   /* sanity checks */
00375   if (pdu->hdr->code == 0) {
00376     if (length != sizeof(coap_hdr_t) || pdu->hdr->token_length) {
00377       debug("coap_pdu_parse: empty message is not empty\n");
00378       goto discard;
00379     }
00380   }
00381 
00382   if (length < sizeof(coap_hdr_t) + pdu->hdr->token_length
00383       || pdu->hdr->token_length > 8) {
00384     debug("coap_pdu_parse: invalid Token\n");
00385     goto discard;
00386   }
00387 
00388 #ifndef WITH_LWIP
00389   /* Copy message id in network byte order, so we can easily write the
00390    * response back to the network. */
00391   memcpy(&pdu->hdr->id, data + 2, 2);
00392 
00393   /* Append data (including the Token) to pdu structure, if any. */
00394   if (length > sizeof(coap_hdr_t)) {
00395     memcpy(pdu->hdr + 1, data + sizeof(coap_hdr_t), length - sizeof(coap_hdr_t));
00396   }
00397   pdu->length = length;
00398  
00399   /* Finally calculate beginning of data block and thereby check integrity
00400    * of the PDU structure. */
00401 #endif
00402 
00403   /* skip header + token */
00404   length -= (pdu->hdr->token_length + sizeof(coap_hdr_t));
00405   opt = (unsigned char *)(pdu->hdr + 1) + pdu->hdr->token_length;
00406 
00407   while (length && *opt != COAP_PAYLOAD_START) {
00408     if (!next_option_safe(&opt, (size_t *)&length)) {
00409       debug("coap_pdu_parse: drop\n");
00410       goto discard;
00411     }
00412   }
00413 
00414   /* end of packet or start marker */
00415   if (length) {
00416     assert(*opt == COAP_PAYLOAD_START);
00417     opt++; length--;
00418 
00419     if (!length) {
00420       debug("coap_pdu_parse: message ending in payload start marker\n");
00421       goto discard;
00422     }
00423 
00424     debug("set data to %p (pdu ends at %p)\n", (unsigned char *)opt, 
00425       (unsigned char *)pdu->hdr + pdu->length);
00426     pdu->data = (unsigned char *)opt;
00427   }
00428 
00429   return 1;
00430 
00431  discard:
00432   return 0;
