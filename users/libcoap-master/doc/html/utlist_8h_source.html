00031 /*
00032  * This file contains macros to manipulate singly and doubly-linked lists.
00033  *
00034  * 1. LL_ macros:  singly-linked lists.
00035  * 2. DL_ macros:  doubly-linked lists.
00036  * 3. CDL_ macros: circular doubly-linked lists.
00037  *
00038  * To use singly-linked lists, your structure must have a "next" pointer.
00039  * To use doubly-linked lists, your structure must "prev" and "next" pointers.
00040  * Either way, the pointer to the head of the list must be initialized to NULL.
00041  *
00042  * ----------------.EXAMPLE -------------------------
00043  * struct item {
00044  *      int id;
00045  *      struct item *prev, *next;
00046  * }
00047  *
00048  * struct item *list = NULL:
00049  *
00050  * int main() {
00051  *      struct item *item;
00052  *      ... allocate and populate item ...
00053  *      DL_APPEND(list, item);
00054  * }
00055  * --------------------------------------------------
00056  *
00057  * For doubly-linked lists, the append and delete macros are O(1)
00058  * For singly-linked lists, append and delete are O(n) but prepend is O(1)
00059  * The sort macro is O(n log(n)) for all types of single/double/circular lists.
00060  */
00061 
00062 /* These macros use decltype or the earlier __typeof GNU extension.
00063    As decltype is only available in newer compilers (VS2010 or gcc 4.3+
00064    when compiling c++ code), this code uses whatever method is needed
00065    or, for VS2008 where neither is available, uses casting workarounds. */
00066 #ifdef _MSC_VER            /* MS compiler */
00067 #if _MSC_VER >= 1600 && defined(__cplusplus)  /* VS2010 or newer in C++ mode */
00068 #define LDECLTYPE(x) decltype(x)
00069 #else                     /* VS2008 or older (or VS2010 in C mode) */
00070 #define NO_DECLTYPE
00071 #define LDECLTYPE(x) char*
00072 #endif
00073 #elif defined(__ICCARM__)
00074 #define NO_DECLTYPE
00075 #define LDECLTYPE(x) char*
00076 #else                      /* GNU, Sun and other compilers */
00077 #define LDECLTYPE(x) __typeof(x)
00078 #endif

00080 /* for VS2008 we use some workarounds to get around the lack of decltype,
00081  * namely, we always reassign our tmp variable to the list head if we need
00082  * to dereference its prev/next pointers, and save/restore the real head.*/
00083 #ifdef NO_DECLTYPE
00084 #define _SV(elt,list) _tmp = (char*)(list); {char **_alias = (char**)&(list); *_alias = (elt); }
00085 #define _NEXT(elt,list,next) ((char*)((list)->next))
00086 #define _NEXTASGN(elt,list,to,next) { char **_alias = (char**)&((list)->next); *_alias=(char*)(to); }
00087 /* #define _PREV(elt,list,prev) ((char*)((list)->prev)) */
00088 #define _PREVASGN(elt,list,to,prev) { char **_alias = (char**)&((list)->prev); *_alias=(char*)(to); }
00089 #define _RS(list) { char **_alias = (char**)&(list); *_alias=_tmp; }
00090 #define _CASTASGN(a,b) { char **_alias = (char**)&(a); *_alias=(char*)(b); }
00091 #else
00092 #define _SV(elt,list)
00093 #define _NEXT(elt,list,next) ((elt)->next)
00094 #define _NEXTASGN(elt,list,to,next) ((elt)->next)=(to)
00095 /* #define _PREV(elt,list,prev) ((elt)->prev) */
00096 #define _PREVASGN(elt,list,to,prev) ((elt)->prev)=(to)
00097 #define _RS(list)
00098 #define _CASTASGN(a,b) (a)=(b)
00099 #endif
