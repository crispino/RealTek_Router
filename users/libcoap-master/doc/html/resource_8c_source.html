00104 static int
00105 match(const str *text, const str *pattern, int match_prefix, int match_substring) {
00106   assert(text); assert(pattern);
00107   
00108   if (text->length < pattern->length)
00109     return 0;
00110 
00111   if (match_substring) {
00112     unsigned char *next_token = text->s;
00113     size_t remaining_length = text->length;
00114     while (remaining_length) {
00115       size_t token_length;
00116       unsigned char *token = next_token;
00117       next_token = memchr(token, ' ', remaining_length);
00118 
00119       if (next_token) {
00120         token_length = next_token - token;
00121         remaining_length -= (token_length + 1);
00122         next_token++;
00123       } else {
00124         token_length = remaining_length;
00125         remaining_length = 0;
00126       }
00127       
00128       if ((match_prefix || pattern->length == token_length) &&
00129             memcmp(token, pattern->s, pattern->length) == 0)
00130         return 1;
00131     }
00132     return 0;
00133   }
00134 
00135   return (match_prefix || pattern->length == text->length) &&
00136     memcmp(text->s, pattern->s, pattern->length) == 0;
00137 }
