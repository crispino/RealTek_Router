00055 void
00056 coap_ticks(coap_tick_t *t) {
00057   unsigned long tmp;
00058 
00059 #ifdef COAP_CLOCK
00060   struct timespec tv;
00061   clock_gettime(COAP_CLOCK, &tv);
00062   /* Possible errors are (see clock_gettime(2)):
00063    *  EFAULT tp points outside the accessible address space.
00064    *  EINVAL The clk_id specified is not supported on this system.
00065    * Both cases should not be possible here.
00066    */
00067 
00068   tmp = SHR_FP(tv.tv_nsec * Q(FRAC, (COAP_TICKS_PER_SECOND/1000000000.0)), FRAC);
00069 #else /* _POSIX_TIMERS */
00070   /* Fall back to gettimeofday() */
00071 
00072   struct timeval tv;
00073   gettimeofday(&tv, NULL);
00074   /* Possible errors are (see gettimeofday(2)):
00075    *  EFAULT One of tv or tz pointed outside the accessible address space.
00076    *  EINVAL Timezone (or something else) is invalid.
00077    * Both cases should not be possible here.
00078    */
00079 
00080   tmp = SHR_FP(tv.tv_usec * Q(FRAC, (COAP_TICKS_PER_SECOND/1000000.0)), FRAC);
00081 #endif /* not _POSIX_TIMERS */
00082 
00083   /* Finally, convert temporary FP representation to multiple of
00084    * COAP_TICKS_PER_SECOND */
00085   *t = tmp + (tv.tv_sec - coap_clock_offset) * COAP_TICKS_PER_SECOND;
00086 }
