00123   const unsigned char hex[] = "0123456789ABCDEF";
00124   unsigned int cnt = 0;
00125   assert(data || len == 0);
00126 
00127   if (buflen == 0) { /* there is nothing we can do here but return */
00128     return 0;
00129   }
00130 
00131   while (len) {
00132     if (!encode_always && isprint(*data)) {
00133       if (cnt+1 < buflen) { /* keep one byte for terminating zero */
00134       *result++ = *data;
00135       ++cnt;
00136       } else {
00137     break;
00138       }
00139     } else {
00140       if (cnt+4 < buflen) { /* keep one byte for terminating zero */
00141     *result++ = '\\';
00142     *result++ = 'x';
00143     *result++ = hex[(*data & 0xf0) >> 4];
00144     *result++ = hex[*data & 0x0f];

00146       } else
00147     break;
00148     }
00149

00151   }
00152 
00153   *result = '\0'; /* add a terminating zero */
